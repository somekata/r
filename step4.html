3. 機械学習モデルの構築

install.packages("caret")
library(caret)

caret での線形回帰モデルの作成

set.seed(123) train_index <- createDataPartition(mtcars$mpg, p = 0.7, list = FALSE) train_data <- mtcars[train_index, ] test_data <- mtcars[-train_index, ] model <- train(mpg ~ ., data = train_data, method = "lm") predictions <- predict(model, test_data) RMSE <- sqrt(mean((predictions - test_data$mpg)^2)) print(paste("RMSE:", round(RMSE, 2)))
