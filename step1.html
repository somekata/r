3. 制御構文

✅ ifelse() の活用

scores <- c(80, 45, 90, 30, 75) results <- ifelse(scores >= 50, "合格", "不合格") print(results)

✅ forループ

for (i in 1:5) { print(paste("回数:", i)) }

✅ whileループ

count <- 1 while (count <= 5) { print(paste("カウント:", count)) count <- count + 1 }

✅ switch() の活用

option <- "B" result <- switch(option, "A" = "選択肢 A", "B" = "選択肢 B", "C" = "選択肢 C", "不明な選択肢") print(result)

library(dplyr) iris %>% filter(Sepal.Length > 5) %>% mutate(Size = ifelse(Sepal.Length >= 6, "大", "小")) %>% head(10)
